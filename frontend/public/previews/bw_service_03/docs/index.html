Authentication

All API requests are authenticated using your API key. Pass it in the Authorization header as a Bearer token.

code
// HTTP Request
GET https://api.payflow.in/v1/orders
Authorization: Bearer pf_live_sk_xxxxxxxxxxxxx
Content-Type: application/json

// Node.js example
const payflow = require('@payflow/node')('pf_live_sk_xxxxxxxxxxxxx')

// Python example
import payflow
payflow.api_key = 'pf_live_sk_xxxxxxxxxxxxx'
client = payflow.Client()
!

Keep your API key secret. Never expose it in client-side code or public repositories.

Create Order

POSThttps://api.payflow.in/v1/v1/orders

Create a new payment order. Returns an order_id that you use to initialise the checkout SDK on the client.

code
// Request
POST https://api.payflow.in/v1/orders
Authorization: Bearer pf_live_sk_xxxxxxxxxxxxx
Content-Type: application/json

{
  "amount": 49900,        // Amount in paise (₹499.00)
  "currency": "INR",
  "receipt": "order_rcptid_1234",
  "notes": {
    "product": "Annual subscription",
    "customer_id": "cust_9812"
  }
}

// Response (200 OK)
{
  "id": "order_P8KvGBTmzW9j4n",
  "entity": "order",
  "amount": 49900,
  "currency": "INR",
  "status": "created",
  "receipt": "order_rcptid_1234",
  "created_at": 1747209600
}

Verify Payment

POSThttps://api.payflow.in/v1/v1/payments/verify

After the customer completes payment, verify the signature on your server before fulfilling the order. Never trust client-side success callbacks alone.

code
// Request
POST https://api.payflow.in/v1/payments/verify
Authorization: Bearer pf_live_sk_xxxxxxxxxxxxx
Content-Type: application/json

{
  "order_id": "order_P8KvGBTmzW9j4n",
  "payment_id": "pay_L4tGRKpxQy7m2v",
  "signature": "9ef4dffbfd84f1318f6739..."
}

// Response (200 OK)
{
  "verified": true,
  "payment_id": "pay_L4tGRKpxQy7m2v",
  "order_id": "order_P8KvGBTmzW9j4n",
  "amount": 49900,
  "method": "upi",
  "status": "captured",
  "settled_at": null
}

// Node.js verification helper
const crypto = require('crypto')

function verifySignature(orderId, paymentId, signature) {
  const body = orderId + '|' + paymentId
  const expected = crypto
    .createHmac('sha256', process.env.PAYFLOW_SECRET)
    .update(body)
    .digest('hex')
  return crypto.timingSafeEqual(
    Buffer.from(expected),
    Buffer.from(signature)
  )
}
!

Always verify on the server using HMAC-SHA256. A mismatched signature means the payment data was tampered with.

Webhooks

PayFlow sends real-time POST requests to your webhook URL for every payment event. Subscribe to events in your dashboard under Settings → Webhooks.

code
// Webhook payload (payment.captured)
{
  "event": "payment.captured",
  "payload": {
    "payment": {
      "entity": {
        "id": "pay_L4tGRKpxQy7m2v",
        "order_id": "order_P8KvGBTmzW9j4n",
        "amount": 49900,
        "currency": "INR",
        "method": "upi",
        "status": "captured",
        "captured_at": 1747209710
      }
    }
  }
}

// Express.js handler
app.post('/webhooks/payflow', express.raw({ type: 'application/json' }), (req, res) => {
  const sig = req.headers['x-payflow-signature']
  const isValid = payflow.webhooks.verify(req.body, sig, process.env.WEBHOOK_SECRET)
  if (!isValid) return res.status(400).send('Invalid signature')

  const event = JSON.parse(req.body)
  if (event.event === 'payment.captured') {
    // Fulfil the order
    fulfillOrder(event.payload.payment.entity.order_id)
  }

  res.json({ received: true })
})
!

Respond with 2xx within 5 seconds. PayFlow retries failed deliveries up to 5 times with exponential backoff.
